 10      
 11     var mz = function() {}
 12     mz.prototype = 
 13 
 14     /**@lends MZ */
 15     {
 16         extend: function(obj) {
 17             var fn = function(options) {
 18                 if(options){
 19                     this.options = options ;
 20                 }
 21             };
 22             fn.prototype = obj;
 23             return fn;
 24         },
 25         /***
 26          *命名空间
 27          *@param {String} namespace 命名的字符串
 28          *@param {Object} obj 要赋予命名空间的对象
 29          *@param {Object} win 默认为window对象
 30          *@example
 31          *MZ.array.removeRepeat([1,2,3,4,5,1,2])
 32          *@returns {Object}
 33          */
 34         namespace: function(namespace, obj, win) {
 35 
 36             var path = namespace.split(".");
 37             var target = win || window;
 38             while (path.length > 0) {
 39                 var p = path.shift();
 40                 if (!target[p]) {
 41                     if (path.length > 0) {
 42                         target[p] = {};
 43                     } else {
 44                         target[p] = obj || {};
 45                     }
 46                 }
 47                 target = target[p];
 48             }
 49             return target;
 50         }
 51     };
 52     window.MZ = new mz();
