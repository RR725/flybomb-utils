  9     /**定义下拉框组件
 10      *@namespace
 11      *@name MZ.dropmenu
 12      */
 13     MZ.namespace("MZ.ui.dropmenu",
 14 
 15         /**@lends MZ.dropmenu */
 16         MZ.extend({
 17 
 18             /**
 19              *初始化入口，插入下拉框的html，绑定事件
 20              *@param {Object} options 可选参数集合
 21              *@param {Number} options.container 下拉框的外层容器，事先写好在html页面中<div id="test"></div>
 22              *@param {String} options.defaultText 默认显示的文字
 23              *@param {String} options.cssClass 自定义样式
 24              *@param {String} options.data 下拉框的数组
 25              *@example
 26                 dropmenu.render({
 27                     defaultText: '选择一',
 28                     container: '#test',
 29                     defaultText: '请选择',
 30                     cssClass: 'xxxxx',
 31                     data: [{
 32                         text: '选项一',
 33                         attrs: {
 34                             id: 3,
 35                             name: 99
 36                         }
 37                     }, {
 38                         text: '选项二',
 39                         attrs: {
 40                             id: 123,
 41                             name: 199
 42                         }
 43                     }]
 44                 });
 45                 <div id="test"></div>
 46                 <script>
 47                     var dropmenu=new MZ.ui.dropmenu();
 48                     dropmenu.render({
 49                         defaultText: '选择一',
 50                         container: '#test',
 51                         defaultText: '请选择',
 52                         cssClass: 'xxxxx',
 53                         data: [{
 54                             text: '选项一',
 55                             attrs: {
 56                                 id: 3,
 57                                 name: 99
 58                             }
 59                         }, {
 60                             text: '选项二',
 61                             attrs: {
 62                                 id: 123,
 63                                 name: 199
 64                             }
 65                         }]
 66                     });
 67                 </script>
 68                 */
 69             render: function(options) {
 70                 var tpl = this._template(options.data);
 71                 var container = options.container;
 72                 var cssClass = options.cssClass ? options.cssClass : '';
 73                 options.id = container.replace('#', '');
 74                 var html = MZ.string.format(tpl.textDown, options);
 75                 this.el = $(container);
 76                 this.el.addClass('ui_dropmeun_dom').addClass(cssClass).html(html);
 77                 this.initEvents(options);
 78             },
 79             /**
 80              *绑定事件，显示下拉框
 81              */
 82             initEvents: function(options) {
 83                 this.showList(options);
 84             },
 85             /**
 86              *选择下拉框里的项，显示在默认框里
 87              */
 88             menuSelect: function(options) {
 89                 var self = this;
 90                 var listId = options.container + 'DropmenuList';
 91                 $(listId).find('li').bind('click', {
 92                     self: self
 93                 }, this.selectEvent);
 94             },
 95 
 96             /**
 97              *选择下拉框里的内容后，填充所选中的项
 98              */
 99             selectEvent: function(ev) {
100                 var self = ev.data.self;
101                 var target = ev.target;
102                 var text = $(target).text();
103                 var attrs = self.attrs;
104                 for (var i in attrs) {
105                     self.el.attr('data-' + i, $(target).attr('data-' + i));
106                 }
107                 self.el.find('.dropmenu_text').html(text);
108             },
109             /**
110              *设置下拉框的位置，最小宽度
111              */
112             setMenuPosition: function(listId, dropmenu) {
113 
114                 var left = this.el.offset().left;
115                 var top = this.el.offset().top;
116                 var height = this.el.height();
117                 var width = dropmenu.width();
118                 $(listId).show().css({
119                     left: left,
120                     top: top + height - 1,
121                     minWidth: width
122                 });
123             },
124             /**
125              *生成下拉框，绑定相关事件
126              */
127             showList: function(options) {
128                 var self = this;
129                 this.el.find('.dropmenu_down').click(function() {
130                     var listId = options.container + 'DropmenuList';
131                     var dropmenu = self.el.find('.ui_dropmenu');
132                     var status = $(listId).is(':visible');
133                     if (status) return; //点击时如果下拉框本身已经是可见状态，会触发全局的globalHide直接隐藏掉，不再处理后面的代码
134                     var tpl = self._template(options.data);
135                     var html = MZ.string.format(tpl.menuList, options);
136                     var exist = $(listId)[0];
137 
138                     if (!exist) {
139                         $('body').append(html);
140                     }
141                     var radius = dropmenu.css('borderRadius');
142                     if (radius && radius != '0px') {
143                         dropmenu.css({
144                             borderRadius: radius + ' ' + radius + ' 0 0'
145                         });
146                     }
147                     self.setMenuPosition(listId, dropmenu);
148                     //绑定全局body点击事件，隐藏下拉框
149                     self._bindGolbalHide(listId, dropmenu, radius);
150                     //绑定事件，选择下拉框里的项
151                     self.menuSelect(options);
152 
153                 });
154             },
155             /**
156              *全局的点击隐藏方法
157              */
158             _bindGolbalHide: function(listId, dropmenu, radius) {
159                 var self = this;
160                 MZ.dom.globalHide({
161                     dom: listId,
162                     status: true,
163                     callback: function() {
164                         if (radius && radius != '0px') {
165                             dropmenu.css({
166                                 borderRadius: radius
167                             });
168                         }
169 
170                         $(listId).find('li').unbind('click', self.selectEvent);
171                     }
172                 });
173             },
174             /**
175              *下拉框的模板
176              */
177             _template: function(data) {
178                 var len = data.length;
179                 var li = '';
180                 for (var i = 0; i < len; i++) {
181                     var attrs = data[i].attrs;
182                     this.attrs = attrs;
183                     var attrsData = '';
184                     for (var o in attrs) {
185                         attrsData += ' data-' + o + '=' + attrs[o];
186                     }
187                     li += '<li' + attrsData + '>' + data[i].text + '</li>';
188                 }
189                 var textDown = ['<div class="ui_dropmenu">',
190                     '<div class="dropmenu_text">{defaultText}</div>',
191                     '<div class="dropmenu_down"><span></span></div>',
192                     '</div>'
193                 ].join('');
194                 var menuList = [
195                     '<div id="{id}DropmenuList" class="dropmenu_list">',
196                     '<ul>' + li + '</ul>',
197                     '</div>'
198                 ].join('');
199                 var json = {
200                     textDown: textDown,
201                     menuList: menuList
202                 };
203                 return json;
204             }
205         }));
