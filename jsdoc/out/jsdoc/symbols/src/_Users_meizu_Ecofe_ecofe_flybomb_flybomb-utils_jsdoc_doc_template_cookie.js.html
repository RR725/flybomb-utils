1 /**
  2  * @fileOverview 读取cookie和设置cookie.
  3  */
  4 /**
  5  *@namespace
  6  *@name cookie
  7  */
  8 const cookie = {
  9 	/**
 10 	 *读取cookie值
 11 	 *@returns {String}
 12 	 */
 13 	get: function(name) {
 14 		var arr = document.cookie.match(
 15 			new RegExp("(^|\\W)" + name + "=([^;]*)(;|$)")
 16 		);
 17 		if (arr != null) return unescape(arr[2]);
 18 		return "";
 19 	},
 20 	/**
 21 	 *@param {Object} options 参数配置
 22 	 *@param {String} options.name cookie的名称
 23 	 *@param {String} options.value cookie的值
 24 	 *@param {String} options.domain cookie访问权限域名，默认为当前域名
 25 	 *@param {String} options.path 默认为 /
 26 	 *@param {Date} options.expries 如果不设置，则默认为会话cookie
 27 	 *@returns {void}
 28 	 */
 29 	set: function(options) {
 30 		var name = options.name;
 31 		var value = options.value;
 32 		var path = options.path || "/";
 33 		var domain = options.domain;
 34 		var expries = options.expries;
 35 		var str = name + "=" + escape(value) + "; ";
 36 		str += "path=" + path + "; ";
 37 		if (domain) str += "domain=" + domain + "; ";
 38 		if (expries) str += "expires=" + expires.toGMTString() + "; ";
 39 		document.cookie = str;
 40 	}
 41 };
 42 module.exports = cookie;
