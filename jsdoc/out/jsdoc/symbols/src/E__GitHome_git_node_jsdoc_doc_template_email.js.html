1 /**
  2  * @fileOverview 定义和邮件地址相关的方法.
  3  */
  4 
  5 (function() {
  6     /**定义和邮件地址相关的方法
  7      *@namespace
  8      *@name MZ.email
  9      */
 10     MZ.namespace('MZ.utils.email',
 11         /**@lends MZ.email */
 12         MZ.extend({
 13 
 14             /**
 15              *获得检测邮件地址的正则表达式(可能来自全局配置)
 16              *@returns {RegExp}
 17              */
 18             getEmailRegex: function() {
 19                 return /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,6}$/i;
 20             },
 21             /**
 22             * 验证邮箱地址是否合法
 23             * @param {string} text 验证的邮箱地址字符串
 24             * @returns {Boolean}
 25             * @example
 26             MZ.email.isEmail("kaifei@meizu.com");//返回true
 27             */
 28             isEmail: function(text) {
 29                 if (typeof text != "string") return false;
 30                 text = $.trim(text);
 31                 var reg = this.getEmailRegex();
 32                 return reg.test(text);
 33             },
 34             /**
 35             * 得到邮箱地址字符中的账号部分。
 36             * @param {String} email 邮件地址字符串
 37             * @returns {String}
 38             * @example
 39             MZ.email.getAccount("kaifei@meizu.com");//返回"kaifei"
 40             */
 41             getAccount: function(email) {
 42                 if (typeof email != "string") return "";
 43                 email = $.trim(email);
 44                 if (this.isEmail(email)) {
 45                     return email.split("@")[0].toLowerCase();
 46                 } else if (this.isEmailAddr(email)) {
 47                     return email.match(/<([^@<>]+)@[^@<>]+>$/)[1].toLowerCase();
 48                 } else {
 49                     return "";
 50                 }
 51             }
 52 
 53         }));
 54 
 55     //定义缩写
 56  
 57     MZ.email = new MZ.utils.email();
 58 })();
