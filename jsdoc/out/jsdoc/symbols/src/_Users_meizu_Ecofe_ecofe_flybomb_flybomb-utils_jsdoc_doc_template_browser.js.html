 16 			 * @field
 17 			 */
 18 			ie: function() {
 19 				if (!!window.ActiveXObject || "ActiveXObject" in window) {
 20 					return true;
 21 				} else {
 22 					return false;
 23 				}
 24 			},
 25 
 26 			/**是否IE6浏览器
 27 			 * @field
 28 			 */
 29 			ie6: function() {
 30 				var ie6 = !-[1] && !window.XMLHttpRequest;
 31 				return ie6;
 32 			},
 33 
 34 			/**是否IE7浏览器
 35 			 * @field
 36 			 */
 37 			ie7: function() {
 38 				return this.ie && /MSIE 7.0/.test(ua);
 39 			},
 40 
 41 			/** 是否IE8浏览器
 42 			 *@field
 43 			 */
 44 			ie8: function() {
 45 				return this.ie && /MSIE 8.0/.test(ua);
 46 			},

145 			/**
146 			 * 是否支持本地存储
147 			 */
148 			storage: function() {
149 				try {
150 					return "undefined" !== typeof window.localStorage;
151 				} catch (ex) {
152 					return false;
153 				}
154 			}
155 		},
156 	/**
157     *判断ua是否包含某个关键字,即：browser.ua.indexOf(keyword) > -1，可以一次判断多个值
158     *@example
159     browser.uaContains(keyword1,keyword2);
160     */
161 	uaContains: function(keyword1, keyword2) {
162 		var result = false;
163 		for (var i = 0; i < arguments.length; i++) {
164 			result = result || this.ua.indexOf(arguments[i]) > -1;
165 		}
166 		return result;
167 	},
168 	/**
169              
170     *初始化，给browser.is赋值
171     */
172 	init: function() {
173 		var is = (this.is = {});
174 		var _is = this._is;
175 		for (var r in _is) {
176 			if (typeof _is[r] == "object") {
177 				is[r] = _is[r].test(ua);
178 			}
179 			if (typeof _is[r] == "function") {
180 				is[r] = _is[r]();
181 			}
182 		}
183 		is.safari = is.safari && !is.chrome;
184 
185 		var _support = this._support;
186 		var support = (this.support = {});
187 		for (var o in _support) {
188 			if (typeof _support[o] == "function") {
189 				support[o] = _support[o];
190 			}
191 		}
192 	}
193 };
