1 /**
  2  * @fileOverview 对html片断进行转义的方法.
  3  */
  4 /**
  5  *@namespace
  6  *@name html
  7  */
  8 const html = {
  9 	html_decodes: {
 10 		"&": "&",
 11 		""": '"',
 12 		"<": "<",
 13 		">": ">",
 14 		" ": " ",
 15 		"'": "'"
 16 	},
 17 	/**
 18 	 *转义html为安全文本
 19 	 *@returns {String}
 20 	 */
 21 	encode: function(str) {
 22 		if (typeof str != "string") return "";
 23 		str = str
 24 			.replace(/&/g, "&")
 25 			.replace(/</g, "<")
 26 			.replace(/>/g, ">")
 27 			.replace(/\"/g, """)
 28 			.replace(/\'/g, "'")
 29 			.replace(/ /g, " ");
 30 		//.replace(/&#([^\;]+);/ig, "&#$1;"); //将井转成相应的汉字“井”
 31 		return str;
 32 	},
 33 	/**
 34 	 *encode的逆函数
 35 	 *@returns {String}
 36 	 */
 37 	decode: function(text) {
 38 		if (typeof text != "string") return "";
 39 		var map = this.html_decodes;
 40 		//多个replace会有bug
 41 		return text.replace(/("|<|>|&| |')/g, function(
 42 			str,
 43 			item
 44 		) {
 45 			return map[item];
 46 		});
 47 		return text;
 48 	}
 49 };
 50 module.exports = html;
