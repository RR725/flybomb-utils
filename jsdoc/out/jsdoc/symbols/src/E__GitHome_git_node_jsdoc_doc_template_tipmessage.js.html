  5 (function($) {
  6     /**定义tips组件
  7      *@namespace
  8      *@name MZ.tipmessage
  9      */
 10     MZ.namespace("MZ.ui.tipmessage",
 11         /**@lends MZ.tipmessage */
 12         MZ.extend({
 13             _getOptions: function(options) {
 14                 var json = {
 15                     pos: options && options.pos ? options.pos : 'top', //默认在顶部显示
 16                     type: options && options.type ? options.type : 'success', //默认为成功的提示
 17                     msg: options && options.message ? options.message : '提示', //默认显示文字“提示”
 18                     delay: options && options.delay ? options.delay : 500,
 19                     id: options && options.id ? options.id : parseInt(Math.random() * 1000000)
 20                 };
 21                 return json;
 22             },
 23             /**
 24              *显示加载中的提示，默认为顶部绿色提示
 25              *@param {Object} options 可选参数集合
 26              *@param {Number} options.delay 延迟多少毫秒自动消失
 27              *@param {String} options.pos 位置，顶部or中间
 28              *@param {String} options.message 提示文本
 29              *@param {String} options.type 类型，成功or失败    fail or success
 30              *@param {String} options.id tip的标识，需要手动关闭tip时设置该参数
 31              *@example
 32                 MZ.tipmessage.show({message:'加载中',delay:1000,pos:'middle',type:'fail'});
 33                 <script>
 34                  function showTipmessage(pos){
 35                    MZ.tipmessage.show({message:'加载中',delay:1000,pos:pos,type:'fail'});
 36                  }
 37                  </script>
 38                  <button onclick="showTipmessage('middle')">tips组件（中间）</button> <button onclick="showTipmessage('top')">tips组件（顶部）</button>
 39              */
 40             show: function(options) {
 41                 var json = this._getOptions(options),
 42                     pos = json.pos,
 43                     type = json.type,
 44                     msg = json.msg,
 45                     delay = json.delay,
 46                     id = json.id;
 47                     this.delay=delay;
 48                 var cssClass = type == 'success' ? 'tipmessage_success' : 'tipmessage_fail';
 49                 var exist1 = $("#tipmsg" + pos)[0];
 50                 var body = $(document.body);
 51                 if (!exist1) { //判断是否存在tipmsg
 52                     $(document.body).append('<div id="tipmsg' + pos + '" class="ui_tipmessage"></div>');
 53                 }
 54                 var el = $("#tipmsg" + pos);
 55                 var exist2 = $("#tipmsg" + id)[0];
 56                 if (!exist2) {
 57                     msg = '<div id="tipmsg' + id + '" class="ui_tipmessage_list">' + msg + '</div>';
 58                     el.append(msg);
 59                 }
 60                 el.show();
 61                 var child = el.find('#tipmsg' + id);
 62                 child.addClass(cssClass).fadeIn(200);
 63 
 64                 var pl = child.css('paddingLeft'),
 65                     pr = child.css('paddingRight'),
 66                     pt = child.css('paddingTop'),
 67                     pb = child.css('paddingBottom');
 68 
 69                 var width = child.width(); //tip的宽度，加上paddingLeft和paddingRight
 70                 var height = child.height(); //tip的高度,加上paddingTop和paddingBottom
 71                 var left = (body.width() - width - (parseInt(pl.replace('px', '')) * 2)) / 2;
 72                 var scrollTop=$(document.body).scrollTop();
 73                 var top = pos == 'top' ? scrollTop : (($(window).height() - height - (parseInt(pt.replace('px', '')) * 2)) / 2)+scrollTop;
 74                 left = $.browser.msie && parseInt($.browser.version) < 8 ? 0 : left;
 75                 el.css({ //设置tip的位置
 76                     left: left + 'px',
 77                     top: top + 'px'
 78                 });
 79                 child.width(width);
 80                 if (delay) {
 81                     this.hide(id);
 82                 }
 83             },
 84 
 85             /**
 86              *隐藏顶部加载中的提示
 87              *@param {String} id 设置tip的ID，用于手动隐藏tip
 88              *@example
 89                 MZ.tipmessage.hide('loginTip');
 90              */
 91             
 92             hide: function(id) {
 93                 //自动消失
 94                 setTimeout(function() {
 95                     $('#tipmsg' + id).fadeOut(200);
 96                 }, this.delay)
 97             }
 98         }));
 99     MZ.tipmessage = new MZ.ui.tipmessage();
100 })(jQuery);
