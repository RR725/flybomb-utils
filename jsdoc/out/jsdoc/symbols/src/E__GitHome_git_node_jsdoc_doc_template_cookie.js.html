1 /**
  2  * @fileOverview 读取cookie和设置cookie.
  3  */
  4 
  5 (function() {
  6     /**读取cookie和设置cookie
  7      *@namespace
  8      *@name MZ.cookie
  9      */
 10     MZ.namespace('MZ.utils.cookie',
 11         /**@lends MZ.cookie */
 12         MZ.extend({
 13 
 14             /**
 15              *读取cookie值
 16              *@returns {String}
 17              */
 18             get: function(name) {
 19                 var arr = document.cookie.match(new RegExp("(^|\\W)" + name + "=([^;]*)(;|$)"));
 20                 if (arr != null) return unescape(arr[2]);
 21                 return "";
 22             },
 23             /**
 24              *@param {Object} options 参数配置
 25              *@param {String} options.name cookie的名称
 26              *@param {String} options.value cookie的值
 27              *@param {String} options.domain cookie访问权限域名，默认为当前域名
 28              *@param {String} options.path 默认为 /
 29              *@param {Date} options.expries 如果不设置，则默认为会话cookie
 30              *@returns {void}
 31              */
 32             set: function(options) {
 33                 var name = options.name;
 34                 var value = options.value;
 35                 var path = options.path || "/";
 36                 var domain = options.domain;
 37                 var expries = options.expries;
 38                 var str = name + "=" + escape(value) + "; ";
 39                 str += "path=" + path + "; ";
 40                 if (domain) str += "domain=" + domain + "; ";
 41                 if (expries) str += "expires=" + expires.toGMTString() + "; ";
 42                 document.cookie = str;
 43             }
 44 
 45         }));
 46 
 47     //定义缩写
 48 
 49     MZ.cookie = new MZ.utils.cookie();
 50 
 51 })();
