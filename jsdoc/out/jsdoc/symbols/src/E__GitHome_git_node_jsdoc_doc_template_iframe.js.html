1 /**
  2  * @fileOverview 和iframe相关的方法，依赖core/namespace,utils/dom.js
  3  */
  4 //require('dom.js');
  5 (function() {
  6     /**和iframe相关的方法
  7      *@namespace
  8      *@name MZ.iframe
  9      */
 10 
 11     MZ.namespace('MZ.utils.iframe',
 12         /**@lends MZ.iframe */
 13         MZ.extend({
 14             /***
 15              *iframe自适应高度，注意：暂时只支持同域名iframe的自适应
 16              *@param {String} id   可选，dom的id名，当iframe里的页面样式设置了overflow:hidden;width:100%;height:100%;
 17                 或者其他拿不到body正确高度的特殊情况下，拿iframe里某个dom节点做为高度设置
 18              *@example
 19              MZ.iframe.autoHeight()
 20              <iframe class="iframe_auto_height" width="100%" scrolling="no" src="../../../out/jsdoc/symbols/src/E__GitHome_git_node_jsdoc_doc_template_namespace.js.html" style="border:0;overflow:hidden;"></iframe>
 21              <script>MZ.iframe.autoHeight()</script>
 22              */
 23             autoHeight: function(id) {
 24                 var self = this;
 25                 this.arr = [];
 26                 this.time = 0;
 27                 this.id = id;
 28                 MZ.iframe._addEvent(window, 'load', MZ.iframe._getIframe);
 29                 //如果二秒钟之内还没有响应，有可能window load事件已失去响应，浏览器的BUG暂时无法解决，重新再执行一次
 30                 //超过3次仍然不执行，抛出异常
 31                 setTimeout(function() {
 32                     if (self.time == 3) {
 33                         throw new Error("请刷新页面重试，window load事件没有响应");
 34                         return;
 35                     }
 36                     if (self.arr.length == 1) { //说明只是执行了第一次的window load，没有回调成功，后面iframe的load事件没有执行,手动再执行一次
 37                         MZ.iframe._getIframe();
 38                     }
 39                     self.time++;
 40                 }, 2000);
 41 
 42             },
 43             _getIframe: function() {
 44                 o = document.getElementsByTagName('iframe');
 45                 for (i = 0; i < o.length; i++) {
 46                     if (/\biframe_auto_height\b/.test(o[i].className)) {
 47                         MZ.iframe._setHeight(o[i]);
 48                         MZ.iframe._addEvent(o[i], 'load', MZ.iframe._getIframe);
 49                     }
 50                 }
 51             },
 52             //每隔一秒重新获取iframe内body的高度
 53             _resizeIframe:function(e){
 54                 var self=this;
 55                 setInterval(function(){
 56                     var height=self.getHeight(e);
 57                     e.height = height;
 58                 },1000);
 59             },
 60             getHeight:function(e){
 61                 if (e.contentDocument) {
 62                     var height = e.contentDocument.body.offsetHeight + 16;
 63                     if (this.id) {
 64                         height = e.contentDocument.getElementById(this.id).offsetHeight + 16;
 65                     }
 66                 } else {
 67                     var height = e.contentWindow.document.body.scrollHeight;
 68                     if (this.id) {
 69                         height = e.contentWindow.document.getElementById(this.id).scrollHeight;
 70                     }
 71                 }
 72                 return height;
 73             },
 74             //设置iframe的高度
 75             _setHeight: function(e) {
 76                 
 77                 var height=this.getHeight(e);
 78                 this._resizeIframe(e);
 79                 e.height = height;
 80             },
 81             _addEvent: function(obj, evType, fn) {
 82                 this.arr.push(1); //每执行一次push一次，用来判断window load事件无响应
 83                 if (obj.addEventListener) {
 84                     obj.addEventListener(evType, fn, false);
 85                     return true;
 86                 } else if (obj.attachEvent) {
 87                     var r = obj.attachEvent("on" + evType, fn);
 88                     return r;
 89                 } else {
 90                     return false;
 91                 }
 92             }
 93 
 94         }));
 95 
 96 
 97     //定义缩写
 98     MZ.iframe = new MZ.utils.iframe();
 99     
100 
101 })();
