1 /**
  2  * @fileOverview 和input相关的方法.
  3  */
  4 /**
  5  *@namespace
  6  *@name input
  7  */
  8 const input = {
  9 	/***
 10              *聚焦去掉提示文字，失焦显示提示文字
 11              *@param {Object}   options               参数集合
 12              *@param {String}   options.id            必选，输入框的id名
 13              *@param {String}   options.text          必选，输入框默认显示的文字
 14              *@param {Array}    options.color         必选，输入框里的文字颜色，输入文字和默认提示文字
 15              *@param {Function} options.focusCallback 可选，focus的回调函数
 16              *@param {Function} options.blurCallback  可选，blur的回调函数
 17              *@example
 18              *input.placeholder({
 19                   id:'#test',
 20                   text:'密码',
 21                   color:['#000','#999'],
 22                   focusCallback:function(){},
 23                   blurCallback:function(){}
 24               })
 25              
 26              */
 27 	placeholder: function(options) {
 28 		var id = $(options.id);
 29 		var color = options.color[1];
 30 		id.val(options.text).css({
 31 			//初始化设置
 32 			color: color
 33 		});
 34 		id.focus(function() {
 35 			//聚焦
 36 			var text = $(this).val();
 37 			if (text == options.text) {
 38 				id.val("").css({
 39 					color: options.color[0]
 40 				});
 41 			}
 42 			if (options.focusCallback) {
 43 				options.focusCallback();
 44 			}
 45 		});
 46 		id.blur(function() {
 47 			//失焦
 48 			var text = $(this).val();
 49 			if ($.trim(text) == "") {
 50 				id.val(options.text).css({
 51 					color: color
 52 				});
 53 			}
 54 			if (options.blurCallback) {
 55 				options.blurCallback();
 56 			}
 57 		});
 58 	},
 59 	/***
 60              *点击文字出现输入框，可修改文字
 61              *@param {Object}   options               参数集合
 62              *@param {String}   options.element       必选，要进行编辑的标签属性  data-toedit
 63              *@param {String}   options.cssClass      必选，控制输入框的样式
 64              *@param {String}   options.type          可选，输入框的类型   input or textarea   默认input
 65              *@param {Function} options.blurCallback  可选，blur的回调函数
 66              *@example
 67              *input.textToEdit({
 68                   element:'data-toedit',
 69                   cssClass:'xxxxxx',
 70                   type:'textarea',
 71                   blurCallback: function(e) {
 72                     var target = $(e.target);
 73                     target.hide();
 74                     target.prev().show().text(target.val());
 75 
 76                     something();
 77                     .......
 78                   }
 79               })
 80              */
 81 	textToEdit: function(options) {
 82 		var self = this;
 83 		var cssClass = options.cssClass ? options.cssClass : "";
 84 		var type = options.type ? options.type : "input";
 85 		var el = $("[" + options.element + "]");
 86 		var len = el.length;
 87 		for (var i = 0; i < len; i++) {
 88 			(function(i) {
 89 				var eli = el.eq(i);
 90 				eli.click(function() {
 91 					var text = $(this).text();
 92 					var parent = $(this).parent();
 93 					var input = parent.find(".input_edit_area");
 94 					var html = self._editTextTemplate(type, text);
 95 					$(this).hide();
 96 					var existInput = input.length;
 97 					if (!existInput) {
 98 						parent.addClass(cssClass).append(html);
 99 						input = parent.find(".input_edit_area");
100 					} else {
101 						input.show().val(text);
102 					}
103 					input.bind("blur", blur);
104 					self.setCursorPos({
105 						dom: input
106 					});
107 
108 					function blur(e) {
109 						input.unbind("blur", blur);
110 						if (options.blurCallback) {
111 							options.blurCallback(e);
112 							return;
113 						}
114 						var target = $(e.target);
115 						target.hide();
116 						target
117 							.prev()
118 							.show()
119 							.text(target.val());
120 					}
121 				});
122 			})(i);
123 		}
124 	},
125 	/**
126 	 *与input.focus()的区别是，这里捕获异常，并且获得焦点后光标默认在文本最后
127 	 *@param {Object} options 参数集合
128 	 *@param {Object} options.dom 获得焦点的输入框
129 	 *@param {Number} options.pos 光标的位置 0:最前,1:选中所有文本,2:最后,默认为2
130 	 */
131 	setCursorPos: function(options) {
132 		var dom = options.dom;
133 		var len = dom.val().length;
134 		options = options || {};
135 		var pos = options.pos === undefined ? 2 : options.pos;
136 		try {
137 			if (pos == 2) {
138 				if (document.all) {
139 					var r = dom[0].createTextRange();
140 					r.moveStart("character", len);
141 					r.collapse(true);
142 					r.select();
143 				} else {
144 					dom[0].setSelectionRange(len, len);
145 					dom.focus();
146 				}
147 			} else if (pos == 1) {
148 				dom.select();
149 			} else {
150 				dom.focus();
151 			}
152 		} catch (e) {
153 			console.log(e);
154 		}
155 	},
156 	_editTextTemplate: function(type, text) {
157 		if (type == "input") {
158 			var html =
159 				'<input class="input_edit_area" type="text" value="' +
160 				text +
161 				'" />';
162 		} else {
163 			var html =
164 				'<textarea class="input_edit_area">' + text + "</textarea>";
165 		}
166 		return html;
167 	}
168 };
169 module.exports = input;
