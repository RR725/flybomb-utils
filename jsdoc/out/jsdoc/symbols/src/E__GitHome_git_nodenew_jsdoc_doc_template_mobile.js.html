1 /**
  2  * @fileOverview 定义和手机相关的方法.
  3  */
  4 
  5 (function() {
  6     /**定义和手机相关的方法
  7      *@namespace
  8      *@name MZ.mobile
  9      */
 10     MZ.namespace('MZ.utils.mobile',
 11         /**@lends MZ.mobile */
 12         MZ.extend({
 13             /***
 14              *获得匹配中国移动手机号的正则(可能来自全局配置)
 15              *@returns {RegExp}
 16              **/
 17             getChinaMobileRegex: function() {
 18                 return new RegExp("^8613[4-9][0-9]{8}$|^8615[012789][0-9]{8}$|^8618[2378][0-9]{8}$|^8614[7][0-9]{8}$");
 19             },
 20             /***
 21              *获得匹配移动手机号的正则(可能来自全局配置)
 22              *@returns {RegExp}
 23              **/
 24             getMobileRegex: function() {
 25                 return new RegExp("^8613[0-9]{9}$|^8615[012356789][0-9]{8}$|^8618[02356789][0-9]{8}$|^8614[7][0-9]{8}$");
 26             },
 27             /***
 28              *检测输入文本是否为一个手机号（中国大陆的手机号），文本必须为纯数字号码
 29              *@param {String} text 要检测的文本
 30              *@returns {Boolean}
 31              **/
 32             isMobile: function(text) {
 33                 text = this.add86(text);
 34                 return this.getMobileRegex().test(text);
 35             },
 36             /***
 37              *检测输入文本是否为中国移动的手机号，文本必须为纯数字号码
 38              *@param {String} text 要检测的文本
 39              *@returns {Boolean}
 40              **/
 41             isChinaMobile: function(text) {
 42                 text = this.add86(text);
 43                 return this.getChinaMobileRegex().test(text);
 44             },
 45 
 46             /**
 47              *给手机号码添加86，如果已存在则不添加
 48              *@returns {String}
 49              */
 50             add86: function(mobile) {
 51                 if (typeof mobile != "string") mobile = mobile.toString();
 52                 return mobile.trim().replace(/^(?:86)?(?=\d{11}$)/, "86");
 53             },
 54             /**
 55              *移除手机号码前的86
 56              *@returns {String}
 57              */
 58             remove86: function(mobile) {
 59                 if (typeof mobile != "string") mobile = mobile.toString();
 60                 return mobile.trim().replace(/^86(?=\d{11}$)/, "");
 61             }
 62         })
 63 
 64     );
 65 
 66     //定义缩写
 67 
 68     MZ.mobile = new MZ.utils.mobile();
 69 })();
