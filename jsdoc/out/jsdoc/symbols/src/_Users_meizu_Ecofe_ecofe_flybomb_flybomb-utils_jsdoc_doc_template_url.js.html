1 /**
  2  * @fileOverview 定义对url进行处理的方法
  3  */
  4 /**
  5  *@namespace
  6  *@name url
  7  */
  8 const url = {
  9 	/**获取Url中的Get参数
 10     *@param {String} key url中的查询参数
 11     *@param {String} url 可选参数，默认是取当前窗口的location.href
 12     *@returns {String}
 13     *@example
 14     var sid = url.queryString("sid");
 15     */
 16 	queryString: function(key, url) {
 17 		url = url || location.search;
 18 		url = url.split(/&|\?/);
 19 		var result = null;
 20 		key = String(key).toLowerCase();
 21 		for (var i = 0; i < url.length; i++) {
 22 			var keyValue = url[i];
 23 			var part = keyValue.split("=");
 24 			if (part[0].toLowerCase() == key) {
 25 				result = part[1];
 26 				break;
 27 			}
 28 		}
 29 		return result;
 30 	},
 31 	/**
 32     @param {String} url 可选参数,要解析的url，默认是取当前窗口的location.href
 33     @returns {Object} 返回{key:value}对应的所有get参数集合的对象
 34     @example
 35     var obj = url.getQueryObj("http://meizu.com/?a=1&b=2");
 36     //返回
 37     {
 38     a:"1",
 39     b:"2"
 40     }
 41     */
 42 	getQueryObj: function(url) {
 43 		var result = {};
 44 		url = url || location.href;
 45 		if (typeof url != "string") {
 46 			throw "参数url必须是字符串类型";
 47 		}
 48 		if (url.indexOf("?") != -1) {
 49 			var search = url.split("?")[1];
 50 			var list = search.split("&");
 51 
 52 			for (var i = 0; i < list.length; i++) {
 53 				var pair = list[i].split("=");
 54 				var key = pair[0];
 55 				var value = pair[1];
 56 				result[key] = value;
 57 			}
 58 		}
 59 		return result;
 60 	},
 61 	/**拼接Url字符串
 62     *@param {String} url 基础地址，可以带？也可以不带？
 63     *@param {Object} queryObj Get查询参数
 64     *@returns {String}
 65     *@example
 66     var url = url.makeUrl("http://meizu.com",{
 67         sid:"xxxxxxx",
 68         key:"yyyyyyy"
 69     });
 70     得到 http://meizu.com/?sid=xxxxxxx&key=yyyyyyy
 71     或者
 72     var url = url.makeUrl("http://meizu.com","a=1&b=2");
 73     得到 http://meizu.com/?a=1&b=2
 74     自动判断是否应该加上"?"
 75     */
 76 	makeUrl: function(url, queryObj) {
 77 		if (url.indexOf("?") == -1) {
 78 			url += "?";
 79 		}
 80 		if (!/\?$/.test(url)) {
 81 			url += "&";
 82 		}
 83 		if (typeof queryObj == "string") {
 84 			url += queryObj;
 85 		} else {
 86 			url += this.urlEncodeObj(queryObj);
 87 		}
 88 		return url;
 89 	},
 90 
 91 	urlEncodeObj: function(queryObj) {
 92 		var arr = [];
 93 		for (var p in queryObj) {
 94 			if (queryObj.hasOwnProperty(p)) {
 95 				arr.push(p + "=" + encodeURIComponent(queryObj[p]));
 96 			}
 97 		}
 98 		return arr.join("&");
 99 	},
100 	/**
101 	 *根据完整的本地路径或者网络路径，获得文件名
102 	 *@returns {String}
103 	 */
104 	getFileName: function(fullpath) {
105 		if (typeof fullpath == "string") {
106 			var url = fullpath.split("?")[0];
107 			var reg = /[^\/\\]+$/;
108 			var m = url.match(reg);
109 			if (m) {
110 				return m[0];
111 			}
112 		}
113 		return "";
114 	},
115 	/**
116 	 *获得小写的文件扩展名，不带.号
117 	 *@returns {String}
118 	 */
119 	getFileExtName: function(fileName) {
120 		if (fileName && fileName.indexOf(".") > -1) {
121 			return fileName
122 				.split(".")
123 				.pop()
124 				.toLowerCase();
125 		}
126 		return "";
127 	},
128 	/**
129 	 *获得文件名，不包括扩展名
130 	 *@returns {String}
131 	 */
132 	getFileNameNoExt: function(filePath) {
133 		var name = this.getFileName(filePath);
134 		return name.replace(/([^.]+)\.[^.]+$/, "$1");
135 	},
136 
137 	/**
138 	 *根据给出的长度截断文件名，显示...，但是保留扩展名
139 	 *@param {String} fileName 文件名
140 	 *@param {Number} maxLength 要截断的最大长度
141 	 *@returns {String} 返回缩略的文件名
142 	 */
143 	getOverflowFileName: function(fileName, maxLength) {
144 		maxLength = maxLength || 25;
145 		fileName = this.getFileName(fileName);
146 		if (fileName.length <= maxLength) {
147 			return fileName;
148 		}
149 		var point = fileName.lastIndexOf(".");
150 		if (point == -1 || fileName.length - point > 5) {
151 			return fileName.substring(0, maxLength - 2) + "…";
152 		}
153 		var pattern = "^(.{" + (maxLength - 4) + "}).*(\\.[^.]+)$";
154 		return fileName.replace(new RegExp(pattern), "$1…$2");
155 	},
156 
157 	/**
158 	 *判断字符串是否为一个url链接
159 	 *@param {String} url 要判断的文本
160 	 *@returns {Boolean}
161 	 */
162 	isUrl: function(url) {
163 		var reg = /^(http|ftp|https):\/\/[\w\-_]+(\.[\w\-_]+)+([\w\-\.,@?^=%&:/~\+#]*[\w\-\@?^=%&amp;/~\+#])?/;
164 		return reg.test(url);
165 	},
166 
167 	/**
168     *根据一个url返回host
169     *@example
170     url.getHost("http://www.meizu.com/g2");//返回 www.meizu.com
171     */
172 	getHost: function(url) {
173 		url = this.removeHttp(url);
174 		var match = url.match(/([^\/]+)/);
175 		if (match) {
176 			return match[1];
177 		} else {
178 			return "";
179 		}
180 	},
181 
182 	/**
183     *移除一个url的协议部分
184     *@example
185     url.removeHttp("http://www.meizu.com/g2");//返回 www.meizu.com/g2
186     */
187 	removeHttp: function(url) {
188 		try {
189 			return url.replace(/^(http|ftp|https|file):\/\//, "");
190 		} catch (e) {
191 			return "";
192 		}
193 	},
194 
195 	/**
196     *根据一个url移除host部分
197     *@example
198     url.removeHost("http://www.meizu.com/g2");//返回 /g2
199     */
200 	removeHost: function(url) {
201 		url = this.removeHttp(url);
202 		return url.replace(/^[^\/]+/, "");
203 	}
204 };
205 module.exports = url;
