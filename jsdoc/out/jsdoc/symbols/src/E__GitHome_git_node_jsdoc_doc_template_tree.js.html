 89             render: function(options) {
 90                 this.level = 1;
 91                 this.createTree(options);
 92             },
 93             initEvents: function(options) {
 94                 var dom = options.container;
 95                 var self = this;
 96                 $(dom).on('click', '.ui_tree' + this.level + ' .ui_tree_icon', function(e) {
 97                     var _this = $(this);
 98                     MZ.events.stopEvent(e);
 99                     var css = _this.parent().parent().attr('class');
100                     var last = _this.parent().attr('data-last');
101                     if (!css || last == '1') {
102                         options.callback && options.callback();
103                         return;
104                     };
105                     css = css.replace('ui_tree', '');
106                     css = parseInt(css);
107                     self.level = css + 1; //上一级的level+1
108                     var nextLevel = _this.find('.ui_tree' + (css + 1));
109                     if (nextLevel.length) { //判断是否已经存在该级目录，如果存在，不再发起ajax请求，直接显示or隐藏
110                         var visible = nextLevel.is(':visible');
111                         if (visible) {
112                             nextLevel.hide();
113                             _this.find('.fold:first').show().find('.unfold:first').hide();
114                         } else {
115                             nextLevel.show();
116                             _this.find('.fold:first').hide().find('.unfold:first').show();
117                         }
118                         return;
119                     }
120 
121                     MZ.tipmessage.show({
122                         message: '加载中',
123                         id: 'loadingTip'
124                     });
125                     var key = self.sourceKey;
126                     var len = key.length;
127                     var data = {};
128                     for (var i = 0; i < len; i++) {
129                         data[key[i]] = _this.parent().attr('tree-' + key[i]); //得到标签上带tree-标识的属性，做为发起ajax请求的参数
130                     }
131                     //result格式：{"code":200,"value":{"data":[{"text":8888,"attr":{"id":"33","name":"fff"}}]},"message":""}
132                     $.ajax({
133                         url: options.ajaxUrl,
134                         data: data,
135                         type: 'get',
136                         dataType: 'json',
137                         contentType:'application/json',
138                         async: false,
139                         success: function(result) {
140                             if (options.ajaxCallback) {
141                                 options.ajaxCallback(_this,result, function(data) {
142                                     options.data = data;
143                                     self.appendSubmenu(options, _this);
144                                 });
145                             } else {
146                                 options.data = result.value.data;
147                                 self.appendSubmenu(options, _this);
148                             }
149                         }
150                     });
151                 });
152             },
153             appendSubmenu: function(options, _this) {
154                 var html = this.template(options);
155                 _this.find('.fold:first').hide().find('.unfold:first').show();
156                 _this.append(html);
157                 MZ.tipmessage.hide('loadingTip'); //请求成功后隐藏tipmessage
158             },
159             createTree: function(options) {
160                 var html = this.template(options);
161                 var dom = options.container;
162                 $(dom).append(html);
163                 this.initEvents(options);
164 
165             },
166             template: function(options) {
167                 var data = options.data;
168                 var len = data.length;
169                 var list = '';
170                 for (var i = 0; i < len; i++) {
171                     var attr = data[i].attr;
172                     var last = data[i].last ? 1 : 0;
173                     var element = '';
174                     this.sourceKey = [];
175                     for (var o in attr) {
176                         element += 'tree-' + o + '="' + attr[o] + '"';
177                         this.sourceKey.push(o);
178                     }
179                     var display = last ? 'none' : 'inline-block';
180                     list += '<div data-last="' + last + '" ' + element + '><span class="ui_tree_icon"><span class="tree_icon_list"><i class="fold" style="display:' + display + ';">+</i><i class="unfold" style="display:' + display + ';">-</i></span>' + data[i].text + '</span></div>';
181                 }
182                 list = '<div style="padding-left:10px" class="ui_tree' + this.level + '">' + list + '</div>';
183                 var html = list;
184                 return html;
185             }
