1 /**
  2  * @fileOverview 对话框.
  3  */
  4 
  5 (function() {
  6     //用于弹出层叠加，最多允许5层叠加
  7     window.MzDialogBzindex = 0; //弹层z-index
  8     window.MzDialogLzindex = 0; //遮罩层z-index
  9     /**对话框
 10      *@namespace
 11      *@name MZ.dialog
 12      */
 13     MZ.namespace('MZ.ui.dialog',
 14         /**@lends MZ.dialog */
 15         MZ.extend({
 16             /**自定义样式   MZ.dialog.cssClass='xxxxx_dialog'; 通过设置自定义样式修改dialog的CSS，默认不设置
 17              * @field
 18              */
 19             cssClass: '',
 20             /**
 21              *对话框的配置参数
 22              *@param {Object} options 所有的参数
 23              *@param {String} options.type 对话框类型，用于生成简单的对话框 'alert' | 'confirm'，复杂的调用base方法
 24              *@param {Number} options.width 对话框 body的宽度
 25              *@param {Number} options.height 对话框 body的高度
 26              *@param {String} options.hdDisplay 对话框 header的显示状态   'none' | ''
 27              *@param {String} options.ftDisplay 对话框 header的显示状态   'none' | ''
 28              *@param {String} options.btnSure 对话框 确认按钮的文字
 29              *@param {String} options.btnCancel 对话框 取消按钮的文字
 30              *@param {String} options.idSure 对话框 确认按钮的id
 31              *@param {String} options.idCancel 对话框 取消按钮的id
 32              *@param {String} options.btnAlign 对话框 按钮的位置显示   'left' | 'right' | 'center'
 33              *@param {String} options.title 对话框 标题
 34              *@param {String} options.content 对话框 内容
 35              *@param {String} options.waiting 点击按钮后ajax请求的等待文字
 36              *@param {String} options.controlClose 是否自己控制对话框的关闭
 37              *@param {Function} options.fnSure 点击确定按钮后触发
 38              *@param {Function} options.fnCancel 点击取消按钮后触发
 39              *@param {Function} options.callback 生成对话框后的回调，用来控制对话框生成后的一些自定义方法
 40              *@param {Function} options.alertCallback 点击按钮后的回调，只用于alert方法
 41              *@param {Function} options.confirmCallback 点击按钮后的回调，只用于confirm方法
 42              *@returns {Object}
 43              */
 44             getOptions: function(options) {
 45                 var self = this;
 46                 var rdm = Math.ceil(Math.random() * 10000000);
 47                 rdm = '_' + rdm;
 48                 this.rdm = rdm;
 49                 return {
 50                     type: options.type || 'alert',
 51                     width: options.width || '',
 52                     height: options.height || '',
 53                     hdDisplay: options.hdDisplay || '',
 54                     ftDisplay: options.ftDisplay || '',
 55                     idSure: options.idSure || '',
 56                     idCancel: options.idCancel || '',
 57                     btnSure: options.btnSure || '确定',
 58                     btnCancel: options.btnCancel || '取消',
 59                     btnAlign: options.btnAlign || 'right',
 60                     title: options.title || '系统提示',
 61                     content: options.content || '系统提示',
 62                     waiting: options.waiting || '',
 63                     controlClose: options.controlClose || true,
 64                     cssClass: this.cssClass,
 65                     fnSure: options.fnSure || function() {
 66                         self.close(rdm);
 67                     },
 68                     fnCancel: options.fnCancel || function() {
 69                         self.close(rdm);
 70                     },
 71                     callback: options.callback || function() {
 72 
 73                     },
 74                     alertCallback: options.alertCallback || function() {
 75 
 76                     },
 77                     confirmCallback: options.confirmCallback || function() {
 78 
 79                     },
 80                     id: options.id || '',
 81                     rdm: rdm || ''
 82                 }
 83 
 84             },
 85             /**
 86              *初始化dialog
 87              *@param {Object} options 对话框的配置参数
 88              *@returns {Object}
 89              */
 90             render: function(options, callback) {
 91                 var data = this.getOptions(options);
 92 
 93                 this.insertHtml(data);
 94                 this.initEvents(data);
 95                 return this;
 96             },
 97             /**
 98              *插入对话框的dom
 99              *@param {Object} data 对话框的配置参数
100              */
101             insertHtml: function(data) {
102                 var self = this;
103                 $('body').append(this.tplHtml(data));
104                 var mzUiDialog = $('#mzUiDialog' + data.rdm);
105                 mzUiDialog.fadeIn();
106                 var callback = data.callback && data.callback(function() {
107                     //ajax生成的DOM，需要在回调里设置对话框位置
108                     self.setDialogPosition(mzUiDialog);
109                 });
110                 this.setDialogPosition(mzUiDialog);
111             },
112             setDialogPosition:function(mzUiDialog){
113                 //设置对话框位置
114                 var height = mzUiDialog.height();
115                 var width = mzUiDialog.width();
116                 mzUiDialog.css({
117                     marginLeft: -width / 2,
118                     marginTop: -height / 2
119                 });
120             },
121             /**
122              *绑定事件 确定 | 取消 | 关闭
123              *@param {Object} data 对话框的配置参数
124              *@param {String} data.rdm 对话框的rdm
125              */
126             initEvents: function(data) {
127                 var self = this;
128                 var rdm = data.rdm;
129                 var dialog = $('.mzui_dialog');
130                 dialog.on('click', '.dialog_sure', function() {
131                     data.fnSure(self, rdm);
132                     data.confirmCallback('ok');
133                     data.alertCallback('ok');
134                     if (data.controlClose) {
135                         self.close(rdm);
136                     }
137                 });
138                 dialog.on('click', '.dialog_cancel', function() {
139                     data.fnCancel(self, rdm);
140                     data.confirmCallback('cancel');
141                     if (data.controlClose) {
142                         self.close(rdm);
143                     }
144                 });
145                 dialog.on('click', '#mzDialogClose' + rdm, function() {
146                     self.close(rdm);
147                 });
148 
149 
150             },
151             /**
152              *对话框通用方法
153              *@param {Object} options 对话框的配置参数
154              *@example
155              MZ.dialog.base({
156                 content: 'fffff',
157                 title: 'gggggg',
158                 type:'confirm',
159                 fnSure: function(dialog,rdm) {
160                     dialog.close(rdm);
161                 },
162                 fnCancel:function(dialog,rdm){
163                     dialog.close(rdm);
164                 }
165              });
166              */
167 
168             base: function(options, callback) {
169                 this.render(options, callback);
170             },
171             /**
172              *对话框alert方法
173              *@param {String} content 对话框内容
174              *@param {String} title 对话框标题
175              *@param {Function} callback 对话框点击确定后的回调
176              *@example
177              MZ.dialog.alert('xxxx', 'uuuuu');
178              */
179             alert: function(content, title, callback) {
180                 var options = {
181                     content: content,
182                     title: title,
183                     width: 400,
184                     alertCallback: callback
185                 };
186                 options.type = 'alert';
187                 this.render(options);
188             },
189             /**
190              *对话框confirm方法
191              *@param {String} content 对话框内容
192              *@param {String} title 对话框标题
193              *@param {Function} callback 对话框点击确定或者取消后的回调
194              *@example
195              MZ.dialog.confirm('xxxx', 'uuuuu',function(action){
196                 console.log(action)//action=='ok' or 'cancel'
197              });
198              */
199             confirm: function(content, title, callback) {
200                 var options = {
201                     content: content,
202                     title: title,
203                     width: 600,
204                     confirmCallback: callback
205                 };
206                 options.type = 'confirm';
207                 this.render(options);
208             },
209             /**
210              *生成对话框的dom
211              *@param {Object} data 对话框的配置参数
212              *@returns {String}
213              */
214             tplHtml: function(data) {
215                 var height = $(document).height();
216                 var type = data.type;
217                 var ftHtml = '';
218                 if (type == 'alert') {
219                     ftHtml = this.ftSureTplHtml();
220                 }
221                 if (type == 'confirm') {
222                     ftHtml = this.ftSureTplHtml() + this.ftCancelTplHtml();
223                 }
224                 MzDialogBzindex = MzDialogBzindex + 9996;
225                 MzDialogLzindex = MzDialogLzindex + 9995;
226                 var html = ['<div id="mzUiDialog{rdm}" class="mzui_dialog {cssClass}" style="width:{width}px; height:{height}px;display:none;z-index:' + MzDialogBzindex + ';">',
227                     '<div class="dialog_hd" style="display:{hdDisplay};">',
228                     '<a href="javascript:;" id="mzDialogClose{rdm}" class="mz_dialog_close">x</a>',
229                     '<h3>{title}</h3>',
230                     '</div>',
231                     '<div class="dialog_bd" id="{id}">',
232                     '<div class="dialog_con modal-body">{content}</div>',
233                     '</div>',
234                     '<div class="dialog_ft" style="text-align:{btnAlign};display:{ftDisplay};">',
235                     ftHtml,
236                     '</div>',
237                     '</div>',
238                     '<div class="mzui_dialog_layer" style="z-index:' + MzDialogLzindex + ';height:' + height + 'px;"></div>'
239                 ].join("");
240                 html = MZ.string.format(html, data);
241                 return html;
242             },
243             /**
244              *生成对话框的确定按钮
245              *@returns {String}
246              */
247             ftSureTplHtml: function() {
248                 var html = [
249                     '<a href="javascript:;" id="{idSure}" class="dialog_sure btn_normal" title="">{btnSure}</a>'
250                 ].join("");
251                 return html;
252             },
253             /**
254              *生成对话框的取消按钮
255              *@returns {String}
256              */
257             ftCancelTplHtml: function() {
258                 var html = [
259                     '<a href="javascript:;" id="{idCancel}" class="dialog_cancel btn_normal" title="">{btnCancel}</a>'
260                 ].join("");
261                 return html;
262             },
263             /**
264              *关闭对话框
265              *@param {String} rdm  对话框的rdm
266              */
267             close: function(rdm) {
268                 $('#mzUiDialog' + rdm).next().fadeOut(100);
269                 $('#mzUiDialog' + rdm).fadeOut(100);
270                 //延时一点remove，关闭之前有可能要获取DOM节点
271                 setTimeout(function() {
272                     $('#mzUiDialog' + rdm).next().remove();
273                     $('#mzUiDialog' + rdm).remove();
274                 }, 200)
275             }
276 
277         }));
278 
279     //定义缩写
280     MZ.dialog = new MZ.ui.dialog();
281     MZ.dialog.cssClass = ''; //自定义UI
282 
283 })();
