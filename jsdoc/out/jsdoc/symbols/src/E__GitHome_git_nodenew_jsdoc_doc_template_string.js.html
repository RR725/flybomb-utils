  5 (function() {
  6     /**和字符串相关的方法
  7      *@namespace
  8      *@name MZ.string
  9      */
 10 
 11     MZ.namespace('MZ.utils.string',
 12         /**@lends MZ.string */
 13 
 14         MZ.extend({
 15             /***
 16             *获得描述性的文件大小文本，如：传入1124，返回1.1KB
 17             *@param {Number} fileSize 必选参数，文件大小
 18             *@param {String} options.byteChar 可选参数,可以把"B"替换为"字节"
 19             *@param {String} options.maxUnit 可选参数,最大单位,目前支持：B|K|M,默认为G
 20             *@param {String} options.comma 可选参数,是否用逗号分开每千单位
 21             *@returns {String}
 22             *@example
 23             //返回1G
 24             MZ.string.getFileSizeText(1024 * 1024 * 1024);
 25             //返回10字节
 26             MZ.string.getFileSizeText(10,{
 27             byteChar:"字节"
 28             });
 29             //返回102400B
 30             MZ.string.getFileSizeText(102400,{
 31             maxUnit:"B"
 32             });
 33             //返回5,000KB
 34             MZ.string.getFileSizeText(1024 * 5000,{
 35             maxUnit:"K",
 36             comma:true
 37             });
 38             */
 39             getFileSizeText: function(fileSize, options) {
 40                 var unit = "B";
 41                 if (!options) {
 42                     options = {};
 43                 }
 44                 if (options.byteChar) {
 45                     unit = options.byteChar; //用"字节"或者"Bytes"替代z最小单位"B"
 46                     if (options.maxUnit == "B") options.maxUnit = unit;
 47                 }
 48                 var maxUnit = options.maxUnit || "G";
 49                 if (unit != maxUnit && fileSize >= 1024) {
 50                     unit = "K";
 51                     fileSize = fileSize / 1024;
 52                     if (unit != maxUnit && fileSize >= 1024) {
 53                         unit = "M";
 54                         fileSize = fileSize / 1024;
 55                         //debugger
 56                         if (unit != maxUnit && fileSize >= 1024) {
 57                             unit = "G";
 58                             fileSize = fileSize / 1024;
 59                         }
 60                     }
 61                     fileSize = Math.ceil(fileSize * 100) / 100;
 62                 }
 63                 if (options.comma) {
 64                     var reg = /(\d)(\d{3})($|\.)/;
 65                     fileSize = fileSize.toString();
 66                     while (reg.test(fileSize)) {
 67                         fileSize = fileSize.replace(reg, "$1,$2$3");
 68                     }
 69                 }
 70                 return fileSize + unit;
 71             },
 72             /**
 73              *截断字符串，并显示省略号
 74              * @param {String} text 必选参数，要截断的字符串。
 75              * @param {Number} maxLength 必选参数，文字长度。
 76              * @param {Boolean} showReplacer 可选参数，截断后是否显示...，默认为true。
 77              *@returns {String}
 78              */
 79             getTextOverFlow: function(text, maxLength, showReplacer) {
 80                 if (text.length <= maxLength) {
 81                     return text;
 82                 } else {
 83                     return text.substring(0, maxLength) + (showReplacer ? "..." : "");
 84                 }
 85             },
 86             getTextOverFlow2: function(text, maxLength, showReplacer) {
 87                 var charArr = text.split("");
 88                 var byteLen = 0;
 89                 var reg = new RegExp("[\x41-\x5A]|[^\x00-\xff]", "g")
 90                 for (var i = 0; i < charArr.length; i++) {
 91                     var cArr = charArr[i].match(reg);
 92                     byteLen += (cArr == null ? 1 : 2)
 93 
 94                     if (byteLen > maxLength) {
 95                         return text.substring(0, i) + (showReplacer ? "..." : "");
 96                     }
 97                 }
 98                 return text;
 99 
100             },
101             /***
102              *格式化字符串，提供数组和object两种方式
103              *@example
104              *MZ.string.format("hello,{name}",{name:"kitty"})
105              *MZ.string.format("hello,{0}",["kitty"])
106              *@returns {String}
107              */
108             format: function(str, arr) {
109                 var reg;
110                 if ($.isArray(arr)) {
111                     reg = /\{([\d]+)\}/g;
112                 } else {
113                     reg = /\{([\w]+)\}/g;
114                 }
115                 return str.replace(reg, function($0, $1) {
116                     var value = arr[$1];
117                     if (value !== undefined) {
118                         return value;
119                     } else {
120                         return "";
121                     }
122                 });
123             },
124             /**
125              * 得到字符串长度，支持中文
126              * <pre>示例：<br>
127              * MZ.string.getBytes("123");
128              * </pre>
129              * @return {字符长度]
130              */
131             getBytes: function(str) {
132                 var cArr = str.match(/[^\x00-\xff]/ig);
133                 return str.length + (cArr == null ? 0 : cArr.length);
134             },
135 
136             /** 
137              * 文本编辑框文字聚焦到最后
138              * <pre>示例：<br>
139              * MZ.string.textFocusEnd(document.getElementById('text'));
140              * </pre>
141              * @param {Object} textObj 必选参数，文本框DOM对象
142              */
143             textFocusEnd: function(textObj) {
144                 if (textObj) {
145                     textObj.focus();
146                     var len = textObj.value.length;
147                     if (document.selection) { //IE
148                         var sel = textObj.createTextRange();
149                         sel.moveStart('character', len);
150                         sel.collapse();
151                         sel.select();
152                     } else if (typeof textObj.selectionStart == 'number' && typeof textObj.selectionEnd == 'number') {
153                         textObj.selectionStart = textObj.selectionEnd = len; //非IE 
154                     }
155                 }
156             }
157 
158 
159         }));
160 
161 
162     //定义缩写
163  
164     MZ.string = new MZ.utils.string();
165 })();
