 33 define(function() {
 34 //>>excludeStart('excludeRequireCss', pragmas.excludeRequireCss)
 35   if (typeof window == 'undefined')
 36     return { load: function(n, r, load){ load() } };
 37 
 38   var head = document.getElementsByTagName('head')[0];
 39 
 40   var engine = window.navigator.userAgent.match(/Trident\/([^ ;]*)|AppleWebKit\/([^ ;]*)|Opera\/([^ ;]*)|rv\:([^ ;]*)(.*?)Gecko\/([^ ;]*)|MSIE\s([^ ;]*)|AndroidWebKit\/([^ ;]*)/) || 0;
 41 
 42   // use <style> @import load method (IE < 9, Firefox < 18)
 43   var useImportLoad = false;
 44 
 45   // set to false for explicit <link> load checking when onload doesn't work perfectly (webkit)
 46   var useOnload = true;
 47 
 48   // trident / msie
 49   if (engine[1] || engine[7])
 50     useImportLoad = parseInt(engine[1]) < 6 || parseInt(engine[7]) <= 9;
 51   // webkit
 52   else if (engine[2] || engine[8])
 53     useOnload = false;
 54   // gecko
 55   else if (engine[4])
 56     useImportLoad = parseInt(engine[4]) < 18;
 57 
 58 //>>excludeEnd('excludeRequireCss')
 59   //main api object
 60   var cssAPI = {};
 61 
 62 //>>excludeStart('excludeRequireCss', pragmas.excludeRequireCss)
 63   cssAPI.pluginBuilder = './css-builder';
 64 
 65   // <style> @import load method
 66   var curStyle, curSheet;
 67   var createStyle = function () {
 68     curStyle = document.createElement('style');
 69     head.appendChild(curStyle);
 70     curSheet = curStyle.styleSheet || curStyle.sheet;
 71   }
 72   var ieCnt = 0;
 73   var ieLoads = [];
 74   var ieCurCallback;
 75 
 76   var createIeLoad = function(url) {
 77     ieCnt++;
 78     if (ieCnt == 32) {
 79       createStyle();
 80       ieCnt = 0;
 81     }
 82     curSheet.addImport(url);
 83     curStyle.onload = function(){ processIeLoad() };
 84   }
 85   var processIeLoad = function() {
 86     ieCurCallback();
 87 
 88     var nextLoad = ieLoads.shift();
 89 
 90     if (!nextLoad) {
 91       ieCurCallback = null;
 92       return;
 93     }
 94 
 95     ieCurCallback = nextLoad[1];
 96     createIeLoad(nextLoad[0]);
 97   }
 98   var importLoad = function(url, callback) {
 99     if (!curSheet || !curSheet.addImport)
100       createStyle();
101 
102     if (curSheet && curSheet.addImport) {
103       // old IE
104       if (ieCurCallback) {
105         ieLoads.push([url, callback]);
106       }
107       else {
108         createIeLoad(url);
109         ieCurCallback = callback;
110       }
111     }
112     else {
113       // old Firefox
114       curStyle.textContent = '@import "' + url + '";';
115 
116       var loadInterval = setInterval(function() {
117         try {
118           curStyle.sheet.cssRules;
119           clearInterval(loadInterval);
120           callback();
121         } catch(e) {}
122       }, 10);
123     }
124   }
125 
126   // <link> load method
127   var linkLoad = function(url, callback) {
128     var link = document.createElement('link');
129     link.type = 'text/css';
130     link.rel = 'stylesheet';
131     if (useOnload)
132       link.onload = function() {
133         link.onload = function() {};
134         // for style dimensions queries, a short delay can still be necessary
135         setTimeout(callback, 7);
136       }
137     else
138       var loadInterval = setInterval(function() {
139         for (var i = 0; i < document.styleSheets.length; i++) {
140           var sheet = document.styleSheets[i];
141           if (sheet.href == link.href) {
142             clearInterval(loadInterval);
143             return callback();
144           }
145         }
146       }, 10);
147     link.href = url;
148     head.appendChild(link);
149   }
150 
151 //>>excludeEnd('excludeRequireCss')
152   cssAPI.normalize = function(name, normalize) {
153     if (name.substr(name.length - 4, 4) == '.css')
154       name = name.substr(0, name.length - 4);
155 
156     return normalize(name);
157   }
158 
159 //>>excludeStart('excludeRequireCss', pragmas.excludeRequireCss)
160   cssAPI.load = function(cssId, req, load, config) {
161 
162     (useImportLoad ? importLoad : linkLoad)(req.toUrl(cssId + '.css'), load);
163 
164   }
165 
166 //>>excludeEnd('excludeRequireCss')
167   return cssAPI;
168 });
