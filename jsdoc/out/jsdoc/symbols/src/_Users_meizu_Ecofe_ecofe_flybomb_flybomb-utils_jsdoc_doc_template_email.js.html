1 /**
  2  * @fileOverview 定义和邮件地址相关的方法.
  3  */
  4 /**
  5  *@namespace
  6  *@name email
  7  */
  8 const email = {
  9 	/**
 10 	 *获得检测邮件地址的正则表达式(可能来自全局配置)
 11 	 *@returns {RegExp}
 12 	 */
 13 	getEmailRegex: function() {
 14 		return /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,6}$/i;
 15 	},
 16 	/**
 17     * 验证邮箱地址是否合法
 18     * @param {string} text 验证的邮箱地址字符串
 19     * @returns {Boolean}
 20     * @example
 21     email.isEmail("kaifei@meizu.com");//返回true
 22     */
 23 	isEmail: function(text) {
 24 		if (typeof text != "string") return false;
 25 		text = $.trim(text);
 26 		var reg = this.getEmailRegex();
 27 		return reg.test(text);
 28 	},
 29 	/**
 30     * 得到邮箱地址字符中的账号部分。
 31     * @param {String} email 邮件地址字符串
 32     * @returns {String}
 33     * @example
 34     email.getAccount("kaifei@meizu.com");//返回"kaifei"
 35     */
 36 	getAccount: function(email) {
 37 		if (typeof email != "string") return "";
 38 		email = $.trim(email);
 39 		if (this.isEmail(email)) {
 40 			return email.split("@")[0].toLowerCase();
 41 		} else if (this.isEmailAddr(email)) {
 42 			return email.match(/<([^@<>]+)@[^@<>]+>$/)[1].toLowerCase();
 43 		} else {
 44 			return "";
 45 		}
 46 	}
 47 };
 48 module.exports = email;
