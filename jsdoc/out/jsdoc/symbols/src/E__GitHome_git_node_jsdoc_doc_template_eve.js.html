  9         has = "hasOwnProperty",

 13         numsort = function (a, b) {
 14             return a - b;
 15         },

 18         events = {n: {}},
 19     /*\
 20      * eve
 21      [ method ]
 22 
 23      * Fires event with given `name`, given scope and other parameters.
 24 
 25      > Arguments
 26 
 27      - name (string) name of the *event*, dot (`.`) or slash (`/`) separated
 28      - scope (object) context for the event handlers
 29      - varargs (...) the rest of arguments will be sent to event handlers
 30 
 31      = (object) array of returned values from the listeners
 32     \*/
 33         eve = function (name, scope) {
 34       name = String(name);
 35             var e = events,
 36                 oldstop = stop,
 37                 args = Array.prototype.slice.call(arguments, 2),
 38                 listeners = eve.listeners(name),
 39                 z = 0,
 40                 f = false,
 41                 l,
 42                 indexed = [],
 43                 queue = {},
 44                 out = [],
 45                 ce = current_event,
 46                 errors = [];
 47             current_event = name;
 48             stop = 0;
 49             for (var i = 0, ii = listeners.length; i < ii; i++) if ("zIndex" in listeners[i]) {
 50                 indexed.push(listeners[i].zIndex);
 51                 if (listeners[i].zIndex < 0) {
 52                     queue[listeners[i].zIndex] = listeners[i];
 53                 }
 54             }
 55             indexed.sort(numsort);
 56             while (indexed[z] < 0) {
 57                 l = queue[indexed[z++]];
 58                 out.push(l.apply(scope, args));
 59                 if (stop) {
 60                     stop = oldstop;
 61                     return out;
 62                 }
 63             }
 64             for (i = 0; i < ii; i++) {
 65                 l = listeners[i];
 66                 if ("zIndex" in l) {
 67                     if (l.zIndex == indexed[z]) {
 68                         out.push(l.apply(scope, args));
 69                         if (stop) {
 70                             break;
 71                         }
 72                         do {
 73                             z++;
 74                             l = queue[indexed[z]];
 75                             l && out.push(l.apply(scope, args));
 76                             if (stop) {
 77                                 break;
 78                             }
 79                         } while (l)
 80                     } else {
 81                         queue[l.zIndex] = l;
 82                     }
 83                 } else {
 84                     out.push(l.apply(scope, args));
 85                     if (stop) {
 86                         break;
 87                     }
 88                 }
 89             }
 90             stop = oldstop;
 91             current_event = ce;
 92             return out.length ? out : null;
 93         };

 96     /*\
 97      * eve.listeners
 98      [ method ]
 99 
100      * Internal method which gives you array of all event handlers that will be triggered by the given `name`.
101 
102      > Arguments
103 
104      - name (string) name of the event, dot (`.`) or slash (`/`) separated
105 
106      = (array) array of event handlers
107     \*/
108     eve.listeners = function (name) {
109         var names = name.split(separator),
110             e = events,
111             item,
112             items,
113             k,
114             i,
115             ii,
116             j,
117             jj,
118             nes,
119             es = [e],
120             out = [];
121         for (i = 0, ii = names.length; i < ii; i++) {
122             nes = [];
123             for (j = 0, jj = es.length; j < jj; j++) {
124                 e = es[j].n;
125                 items = [e[names[i]], e[wildcard]];
126                 k = 2;
127                 while (k--) {
128                     item = items[k];
129                     if (item) {
130                         nes.push(item);
131                         out = out.concat(item.f || []);
132                     }
133                 }
134             }
135             es = nes;
136         }
137         return out;
138     };

140     /*\
141      * eve.on
142      [ method ]
143      **
144      * Binds given event handler with a given name. You can use wildcards “`*`” for the names:
145      | eve.on("*.under.*", f);
146      | eve("mouse.under.floor"); // triggers f
147      * Use @eve to trigger the listener.
148      **
149      > Arguments
150      **
151      - name (string) name of the event, dot (`.`) or slash (`/`) separated, with optional wildcards
152      - f (function) event handler function
153      **
154      = (function) returned function accepts a single numeric parameter that represents z-index of the handler. It is an optional feature and only used when you need to ensure that some subset of handlers will be invoked in a given order, despite of the order of assignment. 
155      > Example:
156      | eve.on("mouse", eatIt)(2);
157      | eve.on("mouse", scream);
158      | eve.on("mouse", catchIt)(1);
159      * This will ensure that `catchIt()` function will be called before `eatIt()`.
160    *
161      * If you want to put your handler before non-indexed handlers, specify a negative value.
162      * Note: I assume most of the time you don’t need to worry about z-index, but it’s nice to have this feature “just in case”.
163     \*/
164     eve.on = function (name, f) {
165     name = String(name);
166     if (typeof f != "function") {
167       return function () {};
168     }
169         var names = name.split(separator),
170             e = events;
171         for (var i = 0, ii = names.length; i < ii; i++) {
172             e = e.n;
173             e = e.hasOwnProperty(names[i]) && e[names[i]] || (e[names[i]] = {n: {}});
174         }
175         e.f = e.f || [];
176         for (i = 0, ii = e.f.length; i < ii; i++) if (e.f[i] == f) {
177             return fun;
178         }
179         e.f.push(f);
180         return function (zIndex) {
181             if (+zIndex == +zIndex) {
182                 f.zIndex = +zIndex;
183             }
184         };
185     };
186     /*\
187      * eve.f
188      [ method ]
189      **
190      * Returns function that will fire given event with optional arguments.
191    * Arguments that will be passed to the result function will be also
192    * concated to the list of final arguments.
193    | el.onclick = eve.f("click", 1, 2);
194    | eve.on("click", function (a, b, c) {
195    |     console.log(a, b, c); // 1, 2, [event object]
196    | });
197      > Arguments
198    - event (string) event name
199    - varargs (…) and any other arguments
200    = (function) possible event handler function
201     \*/
202   eve.f = function (event) {
203     var attrs = [].slice.call(arguments, 1);
204     return function () {
205       eve.apply(null, [event, null].concat(attrs).concat([].slice.call(arguments, 0)));
206     };
207   };

219      [ method ]

221      * Could be used inside event handler to figure out actual name of the event.
222      **
223      > Arguments

225      - subname (string) #optional subname of the event

249     /*\
250      * eve.off
251      [ method ]
252      **
253      * Removes given function from the list of event listeners assigned to given name.
254    * If no arguments specified all the events will be cleared.
255      **
256      > Arguments
257      **
258      - name (string) name of the event, dot (`.`) or slash (`/`) separated, with optional wildcards
259      - f (function) event handler function
260     \*/
261     /*\
262      * eve.unbind
263      [ method ]
264      **
265      * See @eve.off
266     \*/
267     eve.off = eve.unbind = function (name, f) {
268     if (!name) {
269         eve._events = events = {n: {}};
270       return;
271     }
272         var names = name.split(separator),
273             e,
274             key,
275             splice,
276             i, ii, j, jj,
277             cur = [events];
278         for (i = 0, ii = names.length; i < ii; i++) {
279             for (j = 0; j < cur.length; j += splice.length - 2) {
280                 splice = [j, 1];
281                 e = cur[j].n;
282                 if (names[i] != wildcard) {
283                     if (e[names[i]]) {
284                         splice.push(e[names[i]]);
285                     }
286                 } else {
287                     for (key in e) if (e[has](key)) {
288                         splice.push(e[key]);
289                     }
290                 }
291                 cur.splice.apply(cur, splice);
292             }
293         }
294         for (i = 0, ii = cur.length; i < ii; i++) {
295             e = cur[i];
296             while (e.n) {
297                 if (f) {
298                     if (e.f) {
299                         for (j = 0, jj = e.f.length; j < jj; j++) if (e.f[j] == f) {
300                             e.f.splice(j, 1);
301                             break;
302                         }
303                         !e.f.length && delete e.f;
304                     }
305                     for (key in e.n) if (e.n[has](key) && e.n[key].f) {
306                         var funcs = e.n[key].f;
307                         for (j = 0, jj = funcs.length; j < jj; j++) if (funcs[j] == f) {
308                             funcs.splice(j, 1);
309                             break;
310                         }
311                         !funcs.length && delete e.n[key].f;
312                     }
313                 } else {
314                     delete e.f;
315                     for (key in e.n) if (e.n[has](key) && e.n[key].f) {
316                         delete e.n[key].f;
317                     }
318                 }
319                 e = e.n;
320             }
321         }
322     };

324      * eve.once
325      [ method ]
326      **
327      * Binds given event handler with a given name to only run once then unbind itself.
328      | eve.once("login", f);
329      | eve("login"); // triggers f
330      | eve("login"); // no listeners
331      * Use @eve to trigger the listener.
332      **
333      > Arguments
334      **
335      - name (string) name of the event, dot (`.`) or slash (`/`) separated, with optional wildcards
336      - f (function) event handler function
337      **
338      = (function) same return function as @eve.on
339     \*/
340     eve.once = function (name, f) {
341         var f2 = function () {
342             eve.unbind(name, f2);
343             return f.apply(this, arguments);
344         };
345         return eve.on(name, f2);
346     };
